> $ sudo apt install git

> $ git clone https://github.com/DR4X-c0d3r/d_bCRYPT.git

> $ cd d_bCRYPT

> $ pip install -r requirements.txt

> $ chmod +x d_bCRYPT.py | sudo cp d_bCRYPT.py /usr/bin/d_bCRYPT
